// Contact form submission endpoint

    db.run(
        `INSERT INTO contact_messages (name, email, subject, message, createdAt)
         VALUES (?, ?, ?, ?, ?)`,
        [name, email, subject, message, createdAt],
        function(err) {

            }
            res.json(rows);
        }
    );
